  内容：外部中断0来控制一个循环移位的操作，外部中断0：中断信号从单片机的P3.2引脚输入

#include <REGX52.H>
unsigned char temp; // 声明一个无符号字符型变量 temp
void int0_intle() interrupt 0{ // 定义外部中断0的中断服务程序
    P2 = temp; // 将 temp 的值写入端口 P2
    temp = temp >> 1; // 将 temp 的值右移一位
}
void main() { // 主函数
    IE = 0x81; // 启用外部中断0，并使能总中断
    IT0 = 1; // 设置外部中断0为下降沿触发方式
    temp = 0x01; // 初始化 temp 的值为 00000001
    while (1); // 无限循环
}
