#include<reg52.h> 
void Delay(unsigned int xms)		//@12.000MHz
{
	while (xms--) {
		unsigned char i, j;

		i = 2;
		j = 239;
		do
		{
			while (--j);
		} while (--i);
	}
}
void main()
{
	while (1)
	{
		P2 = 0xAA;		//D1,D3,D5,D7闪烁
		Delay(200);      //调用延时程序；更改延时数字可以更改延时长度；
		P2 = 0xFF;	
		Delay(200);
	}
}
